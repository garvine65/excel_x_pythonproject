
#Data Loading & Prep (Steps 1 & 2)
import pandas as pd

# Load the data
df = pd.read_excel('inventory.xlsx')

print("Available columns after loading data:")
print(df.columns)

# Calculations
df['Revenue'] = df['Quantity'] * df['Selling Price'] * (1 - df['Discount'] / 100)
df['Cost'] = df['Quantity'] * df['Unit Cost'] # Corrected column name
df['Profit'] = df['Revenue'] - df['Cost']
df['Profit_Margin'] = (df['Profit'] / df['Revenue']) * 100

# Show the first few rows to verify
df.head()

# General Stats
print(df[['Revenue', 'Profit', 'Profit_Margin']].describe())

# Category-wise Stats
category_summary = df.groupby('Category').agg({
    'Profit_Margin': 'mean',
    'Revenue': 'sum',
    'Product ID': 'count'
})
